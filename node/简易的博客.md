
## 一、需求分析

前端展示：

1. 分类导航栏

2. 上一页/下一页

3. 文章内容，包含发表时间

4. 允许评论，上一页/下一页

5. 管理员登录

后台管理：

1. 为导航添加新类别

2. 添加新文章，编辑器允许 markdown 语法

3. 管理评论

## 七、静态资源

采用前后端分离开发，静态资源存放在 `/veiws/static/` 目录下

```javascript
app.use("/static", express.static(__dirname + '/static'));
```

`express.static`：表示当遇到静态资源的请求，都调用 `express.static` 方法。而上述代码表示当前端请求的路径以 `static` 开始，直接返回对应的目录下的静态文件。

[更具体的静态资源处理](https://github.com/hoanFir/blogs/blob/master/node/%E7%AE%80%E6%98%93%E7%9A%84%E9%9D%99%E6%80%81%E8%B5%84%E6%BA%90%E6%9C%8D%E5%8A%A1%E5%99%A8.md)

## 九、用户登录

routers/main.js

```

var express = require('express');

var router = express.Router();

// response 格式化
var responseData = null;
router.use(function(req,res,next){
    responseData={
        code: 0,
        message: ''
    }
    next();
});

router.post('/login', function(req, res, next) {
  var username = req.body.username;
  var password = req.body.password;
  
  //由于只有一个管理员，因此写死
  if(username === 'hoanfir' && password === 'hoanfir_password') {
    responseData.code = 200;
    responseData.message = '登录成功';
    res.json(responseData);
    
    return;  
  } else {
    responseData.code = 400;
    responseData.message='用户名或密码错误';
    res.json(responseData);
    
    return;
  }
})


module.exports = router;

```


## 十、简单 cookie 设置

index.js

```javascript

var express = require('express');
var Cookies = require('cookies');

var PORT = 9001;

var app = express();

app.listen(PORT);

console.log('server running at port: ' + PORT + '.');


app.use(function(req, res) {
  req.cookie = new Cookies(req, res);
  
  //解析cookie信息把它由字符转化为对象
  if(req.cookies.get('userInfo')) {
    try {
      req.userInfo = JSON.parse(req.cookie.get('userInfo'));
    } catch(e) {}
  }
  
  next();
})


app.use("/", require('./routers/admin'));
app.use("/api", require('./routers/api'));
app.use("/admin", require('./routers/admin'));

```
